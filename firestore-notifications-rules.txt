// Add these rules to your firestore.rules file

// Admin Notifications Collection
match /admin_notifications/{notificationId} {
  // Allow any authenticated user (including mobile app users) to create notifications
  // This allows the mobile app to notify admins about events
  allow create: if request.auth != null;
  
  // Only admins can read notifications
  // Check if the user exists in adminUsers collection
  allow read: if request.auth != null && 
    exists(/databases/$(database)/documents/adminUsers/$(request.auth.uid));
  
  // Only admins can update notifications (mark as read)
  allow update: if request.auth != null && 
    exists(/databases/$(database)/documents/adminUsers/$(request.auth.uid));
  
  // Only admins can delete notifications
  allow delete: if request.auth != null && 
    exists(/databases/$(database)/documents/adminUsers/$(request.auth.uid));
}
